define

// define a module with function
hexjs.define( id?, dependencies?, factory );

hexjs.define( 'module-id', ['module-a'], function( a, require, exports, module ){
    //todo something
} );

hexjs.define( 'module-id', ['module-a','module-b'], function( a, b, require, exports, moudle ){
    //todo something
} );

hexjs.define( 'module-id', function( require, exports, module ){
    //todo something
} );
            
// define a module with object, this object contains the initialization function named 'init', require that the module is the first implementation of the 'init' function and returns its results, if there is no 'init' function, it returns the object
hexjs.define( 'module-id', {
    init: function( require, exports, module ){
        //todo something
    },
    _other: function(){}
} );

hexjs.define( 'module-id', ['module-a','module-b'], {
    init: function( a, b, require, exports, module ){
        //todo something
    },
    _other: function(){}
} );
            
// define an anonymous module
hexjs.define( function( require, exports, module ){
    //todo something
} );

hexjs.define( ['module-a','module-b'], function( a, b, require, exports, module ){
    //todo something
} );

register

// register a module, and it will be executed after ready
hexjs.register('module-id');
            
// register a module, and it will be executed now
// add a tilde(~) before the module name
hexjs.register('~module-id');
            
// register an anonymous module, and it will be executed after ready
hexjs.define( function( require, exports, module ){
    //todo something
} ).register();
            
// register an anonymous module, and it will be executed now
hexjs.define( function( require, exports, module ){
    //todo something
} ).register('~');
            
// register many modules
hexjs.register( ['module-a', 'module-b', '~module-c'] );

exports

// output API from exports
hexjs.define( 'module-a', function( require, exports, module ){
    exports.a = new Date().getTime(); // --> 1314289663337
} );
            
// output API from return, output object, function, array
hexjs.define( 'module-b', function( require, exports, module ){
    return {
        b: 'bbb'
    }
} );
            
// if 'return' exists, 'exports' will failure
hexjs.define( 'module-c', function( require, exports, module ){
    exports.c = '111';
    exports.d = '222';
    return {
        c: 'ccc'
    }
    // c --> 'ccc', d --> undefined
} );
            
// not covered the exports, need to output an instance, use 'return' or assign to 'module.exports'
hexjs.define( 'module-d', function( require, exports, module ){
    exports = {
        a: '111'
    };
} );

require

// require module-a
hexjs.define( 'module-e', ['moddule-a'], function( A, require, exports, module ){
    var a = require('module-a');
    // A === a --> true
    // A.a --> '1314289663337'
    // a.a --> '1314289663337'
} );
            
// set the re-run option is 'true', the module will executed again, and covering the original results
hexjs.define( 'module-f', function( require, exports, module ){
    var a = require('module-a', true);
    // a.a --> '1314289701153'
} );

module

module.id

// module.id returns the current module name
hexjs.define( 'module-g', function( require, exports, module ){
    console.info( module.id ); // --> module-g
} );

module.exports

// output API from module.exports
hexjs.define( 'module-h', function( require, exports, module ){
    module.exports = {
        a: 'aaa'
    };
} );

module.constructor

// add some method or property to the module constructor
hexjs.define( 'module-constructor', function( require, exports, module ){
    var M = module.constructor;
    
    M.prototype.getId = function(){
        return module.id;
    };
} );

hexjs.define(function(require, exports, module ){
    require('module-constructor');
    console.info( module.getId() ); // --> 'module-constructor'
} ).register();

noConflict

hexjs.noConflict()

// you can use 'myHexjs' as 'hexjs'
var myHexjs = hexjs.noConflict();
myHexjs.define( 'module-name', function(){ ... } );
myHexjs.register('module-name');

hexjs.noConflict(false)

// you can use 'define' as 'hexjs.define' and use 'register' as 'hexjs.register'
hexjs.noConflict(false);
define( 'module-name', function(){ ... } );
register('module-name');

hexjs.noConflict(true)

// free up 'define' and 'register'
hexjs.noConflict(true);
